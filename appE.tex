\chapter{СПЕЦИФИКАЦИЯ ГРАФА АГЕНТОВ LANGGRAPH}

В~приложении представлена текстовая спецификация графа агентов
ИИ-методиста: ноды, основные переходы и~точки участия человека.

\section*{Д.1 --- Ноды графа}

\begin{table}[h!]
  \caption{Ноды графа агентов ИИ-методиста}
  \label{tab:langgraph-nodes}
  \centering
  \begin{tabular}{|p{3.5cm}|p{2.5cm}|p{5.5cm}|}
  \hline
  \textbf{Нода} & \textbf{Тип} & \textbf{Описание} \\
  \hline
  target\_analysis & agent & Анализ целевой аудитории и~контекста использования курса \\
  \hline
  course\_structure & agent & Подготовка структуры курса по~выбранной методологии \\
  \hline
  human\_review & interrupt & Ручное подтверждение или~коррекция структуры \\
  \hline
  content\_creator & agent & Подготовка черновиков уроков с~использованием поиска по~корпусу \\
  \hline
  fact\_check & agent & Проверка утверждений относительно исходного документа \\
  \hline
  media\_gen & agent & Иллюстрации и~TTS-материалы \\
  \hline
  final\_review & agent & Итоговая проверка качества и~сборка отчёта \\
  \hline
  \end{tabular}
\end{table}

\section*{Д.2 --- Основные переходы}

\begin{table}[h!]
  \caption{Ключевые переходы в графе LangGraph}
  \label{tab:langgraph-edges}
  \centering
  \begin{tabular}{|p{3.2cm}|p{3.2cm}|p{5cm}|}
  \hline
  \textbf{Откуда} & \textbf{Куда} & \textbf{Условие перехода} \\
  \hline
  target\_analysis & course\_structure & исходный документ разобран и~аудитория определена \\
  \hline
  course\_structure & human\_review & структура курса построена и~доступна для~проверки \\
  \hline
  human\_review & content\_creator & структура утверждена пользователем \\
  \hline
  content\_creator & fact\_check & черновики уроков подготовлены и~требуют верификации \\
  \hline
  fact\_check & media\_gen & фактологический порог пройден \\
  \hline
  media\_gen & final\_review & медиаартефакты готовы \\
  \hline
  final\_review & export & итоговый пакет курса признан валидным \\
  \hline
  \end{tabular}
\end{table}

\section*{Д.3 --- Human-in-the-loop точки}

\begin{itemize}
  \item подтверждение структуры курса перед генерацией полного контента;
  \item ручной разбор спорных кейсов факт-чекинга;
  \item финальное решение о~публикации курса в~LMS.
\end{itemize}
